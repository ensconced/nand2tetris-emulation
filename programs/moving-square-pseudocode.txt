word_offset = 0

loop {
  // clear screen
  word_idx = 0
  loop {
    if word_idx == 8192 break;
    M[SCREEN + word_idx] = 0
    word_idx += 1
  }

  word_idx = word_offset;
  loop {
    if word_idx - word_offset == 512 break
    SCREEN[word_idx] = 0xffff
    word_idx += 32
  }

  word_offset = (word_offset + 32) & 8191
}
