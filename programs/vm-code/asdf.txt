function left(wordidx, bitidx) {
  if (bitidx == 15) {
    return {
      wordidx: wordidx is even ? wordidx + 1 : wordidx - 1,
      bitidx: 0
    }
  } else {
    return {
      wordidx: wordidx,
      bitidx: bitidx + 1
    }
  }
}

function right(wordidx, bitidx) {
  if (bitidx == 0) {
    return {
      wordidx: wordidx is even ? wordidx + 1 : wordidx - 1,
      bitidx: 15
    }
  } else {
    return {
      wordidx: wordidx,
      bitidx: bitidx - 1
    }
  }
}

function up(wordidx, bitidx) {
  return {
    wordidx: (wordidx + 30) % 32;
    bitidx: bitidx,
  }
}

function down(wordidx, bitidx) {
  return {
    wordidx: (wordidx + 2) % 32;
    bitidx: bitidx,
  }
}
